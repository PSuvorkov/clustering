# Visualize the clustering
def plot_clustering(X_red, labels, y, title=None):
    x_min, x_max = np.min(X_red, axis=0), np.max(X_red, axis=0)
    X_red = (X_red - x_min) / (x_max - x_min)

    plt.figure(figsize=(6, 4))
    for i in range(X_red.shape[0]):
        plt.text(X_red[i, 0], X_red[i, 1], str(y[i]),
                 color=plt.cm.nipy_spectral(labels[i] / 10.),
                 fontdict={'weight': 'bold', 'size': 9})

    plt.xticks([])
    plt.yticks([])
    if title is not None:
        plt.title(title, size=17)
    plt.axis('off')
    plt.tight_layout(rect=[0, 0.03, 1, 0.95])

def flat(filename):
    data = pd.read_excel(filename)
    X = data.iloc[:,1:].to_numpy()
    clustering = AgglomerativeClustering(linkage='ward', n_clusters=7)
    clustering.fit(X)
    plot_clustering(X, clustering.labels_, data['Unnamed: 0'], "%s linkage" % 'ward')
    plt.show()
    data['ward'] = clustering.labels_
    data.to_excel(f"{filename}_agglomerative_ward.xlsx")

def non_flat(filename):
    data = pd.read_excel('  _0803___e_max_e.xlsx')
    print('MEAN SHIFT')
    X = data.iloc[:,1:].to_numpy()
    # #############################################################################
    # Compute clustering with MeanShift

    # The following bandwidth can be automatically detected using
    bandwidth = estimate_bandwidth(X, quantile=0.2, n_samples=402)

    ms = MeanShift(bandwidth=bandwidth, bin_seeding=True)
    ms.fit(X)
    labels = ms.labels_
    cluster_centers = ms.cluster_centers_

    labels_unique = np.unique(labels)
    n_clusters_ = len(labels_unique)

    print("number of estimated clusters : %d" % n_clusters_)

    # #############################################################################
    # Plot result

    plt.figure(1)
    plt.clf()

    colors = cycle('bgrcmykbgrcmykbgrcmykbgrcmyk')
    for k, col in zip(range(n_clusters_), colors):
        my_members = labels == k
        cluster_center = cluster_centers[k]
        plt.plot(X[my_members, 0], X[my_members, 1], col + '.')
        plt.plot(cluster_center[0], cluster_center[1], 'o', markerfacecolor=col,
                 markeredgecolor='k', markersize=14)
    plt.title('Estimated number of clusters: %d' % n_clusters_)
    plt.show()
    data['MeanShift'] = labels
    print('AGGLOMERATIVE AVERAGE')
    clustering = AgglomerativeClustering(linkage='average', n_clusters=7)
    clustering.fit(X)
    plot_clustering(X, clustering.labels_, data['Unnamed: 0'], "%s linkage" % 'average')
    plt.show()
    data['agglomerative_average'] = clustering.labels_
    data.to_excel(f"{filename}_mean_shift_agglomerative_average.xlsx")
